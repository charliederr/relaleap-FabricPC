     487 +def _init_base_params(
     488 +    key: jax.Array,
     489 +    *,
     490 +    vocab_size: int,
     491 +    seq_len: int,
     492 +    hidden_dim: int,
     493 +    layers: int,
     494 +) -> dict[str, Any]:
     495 +    keys = jax.random.split(key, layers + 3)
     496 +    params = {
     497 +        "token_embedding": jax.random.normal(
     498 +            keys[0], (vocab_size, hidden_dim), dtype=jnp.float32
     499 +        )
     500 +        * jnp.float32(0.02),
     501 +        "position_embedding": jnp.zeros((seq_len, hidden_dim), dtype=jnp.float32),
     502 +        "layers": tuple(_init_encoder_layer(k, hidden_dim) for k in keys[1 : layers + 1]),
     503 +        "final_ln_gamma": jnp.ones((hidden_dim,), dtype=jnp.float32),
     504 +        "final_ln_beta": jnp.zeros((hidden_dim,), dtype=jnp.float32),
     505 +        "lm_head": jax.random.normal(
     506 +            keys[-1], (hidden_dim, vocab_size), dtype=jnp.float32
     507 +        )
     508 +        * jnp.float32(1.0 / np.sqrt(hidden_dim)),
     509 +    }
     510 +    return params
     511 +
     512 +
     513 +def _init_encoder_layer(key: jax.Array, hidden_dim: int) -> dict[str, jnp.ndarray]:
     514 +    keys = jax.random.split(key, 6)
     515 +    ff_dim = hidden_dim * 4
     516 +    scale = jnp.float32(1.0 / np.sqrt(hidden_dim))
     517 +    ff_scale = jnp.float32(1.0 / np.sqrt(ff_dim))
     518 +    return {
     519 +        "W_q": jax.random.normal(keys[0], (hidden_dim, hidden_dim), dtype=jnp.float32) * scale,
     520 +        "W_k": jax.random.normal(keys[1], (hidden_dim, hidden_dim), dtype=jnp.float32) * scale,
     521 +        "W_v": jax.random.normal(keys[2], (hidden_dim, hidden_dim), dtype=jnp.float32) * scale,
     522 +        "W_o": jax.random.normal(keys[3], (hidden_dim, hidden_dim), dtype=jnp.float32) * scale,
     523 +        "W_ff1": jax.random.normal(keys[4], (hidden_dim, ff_dim), dtype=jnp.float32) * scale,
     524 +        "W_ff2": jax.random.normal(keys[5], (ff_dim, hidden_dim), dtype=jnp.float32) * ff_scale,
     525 +        "b_q": jnp.zeros((hidden_dim,), dtype=jnp.float32),
     526 +        "b_k": jnp.zeros((hidden_dim,), dtype=jnp.float32),
     527 +        "b_v": jnp.zeros((hidden_dim,), dtype=jnp.float32),
     528 +        "b_o": jnp.zeros((hidden_dim,), dtype=jnp.float32),
     529 +        "b_ff1": jnp.zeros((ff_dim,), dtype=jnp.float32),
     530 +        "b_ff2": jnp.zeros((hidden_dim,), dtype=jnp.float32),
     531 +        "ln1_gamma": jnp.ones((hidden_dim,), dtype=jnp.float32),
     532 +        "ln1_beta": jnp.zeros((hidden_dim,), dtype=jnp.float32),
     533 +        "ln2_gamma": jnp.ones((hidden_dim,), dtype=jnp.float32),
     534 +        "ln2_beta": jnp.zeros((hidden_dim,), dtype=jnp.float32),
     535 +    }

     537 +
     538 +def _encode(base_params: dict[str, Any], input_ids: jnp.ndarray) -> jnp.ndarray:
     539 +    seq_len = input_ids.shape[1]
     540 +    hidden = (
     541 +        base_params["token_embedding"][input_ids]
     542 +        + base_params["position_embedding"][:seq_len]
     543 +    )
     544 +    for layer in base_params["layers"]:
     545 +        hidden = _encoder_layer_forward(layer, hidden)
     546 +    return layer_norm_last(hidden, base_params["final_ln_gamma"], base_params["final_ln_beta"])
     547 +
     548 +
     549 +def _encoder_layer_forward(layer: dict[str, jnp.ndarray], hidden: jnp.ndarray) -> jnp.ndarray:
     550 +    batch_size, seq_len, hidden_dim = hidden.shape
     551 +    num_heads = 4
     552 +    head_dim = hidden_dim // num_heads
     553 +
     554 +    q = jnp.matmul(hidden, layer["W_q"]) + layer["b_q"]
     555 +    k = jnp.matmul(hidden, layer["W_k"]) + layer["b_k"]
     556 +    v = jnp.matmul(hidden, layer["W_v"]) + layer["b_v"]
     557 +
     558 +    q = q.reshape(batch_size, seq_len, num_heads, head_dim).transpose(0, 2, 1, 3)
     559 +    k = k.reshape(batch_size, seq_len, num_heads, head_dim).transpose(0, 2, 1, 3)
     560 +    v = v.reshape(batch_size, seq_len, num_heads, head_dim).transpose(0, 2, 1, 3)
     561 +    scores = jnp.matmul(q, jnp.swapaxes(k, -1, -2)) / jnp.sqrt(jnp.float32(head_dim))
     562 +    attn = jax.nn.softmax(scores, axis=-1)
     563 +    attn_out = jnp.matmul(attn, v).transpose(0, 2, 1, 3).reshape(
     564 +        batch_size, seq_len, hidden_dim
     565 +    )
     566 +    attn_out = jnp.matmul(attn_out, layer["W_o"]) + layer["b_o"]
     567 +    hidden = layer_norm_last(hidden + attn_out, layer["ln1_gamma"], layer["ln1_beta"])
     568 +    ff = gelu(jnp.matmul(hidden, layer["W_ff1"]) + layer["b_ff1"])
     569 +    ff = jnp.matmul(ff, layer["W_ff2"]) + layer["b_ff2"]
     570 +    return layer_norm_last(hidden + ff, layer["ln2_gamma"], layer["ln2_beta"])
     571 +
     572 +
     573 +def _decode(base_params: dict[str, Any], hidden: jnp.ndarray) -> jnp.ndarray:
     574 +    return jnp.matmul(hidden, base_params["lm_head"])

     577 +def _base_forward(base_params: dict[str, Any], input_ids: jnp.ndarray) -> jnp.ndarray:
     578 +    return _decode(base_params, _encode(base_params, input_ids))

     581 +def _forward_with_residual(
     582 +    base_params: dict[str, Any],
     583 +    residual_params: NodeParams,
     584 +    input_ids: jnp.ndarray,
     585 +    *,
     586 +    num_columns: int,
     587 +    top_k: int,
     588 +    support_router: str,

     591 +    settled, _ = residual_forward(

     593 +        hidden,
     594 +        num_columns=num_columns,

     596 +        support_router=support_router,
     597 +    )
     598 +    return _decode(base_params, settled)

     601 +def forward_with_hep_alpha(
     602 +    base_params: dict[str, Any],
     603 +    residual_params: NodeParams,
     604 +    input_ids: jnp.ndarray,
     605 +    *,
     606 +    pc_steps: int,
     607 +    hep_alpha: float,
     608 +    num_columns: int,
     609 +    top_k: int,
     610 +    support_router: str,
     611 +    pinned_support: bool = False,
     612 +    hep_update_clip_norm: float | None = None,
     613 +    hep_settling_objective: str = "residual_adapter",
     614 +    targets: jnp.ndarray | None = None,
     615 +    vocab_size: int | None = None,
     616 +) -> jnp.ndarray:
     617 +    if pc_steps < 1:
     618 +        raise ValueError("pc_steps must be at least 1")
     619 +    if hep_alpha < 0.0 or hep_alpha > 1.0:
     620 +        raise ValueError("hep_alpha must be between 0.0 and 1.0")
     621 +    if hep_update_clip_norm is not None and hep_update_clip_norm <= 0.0:
     622 +        raise ValueError("hep_update_clip_norm must be positive when set")
     623 +    if hep_settling_objective == "supervised_ce_gradient" and (
     624 +        targets is None or vocab_size is None
     625 +    ):
     626 +        raise ValueError("supervised_ce_gradient HEP settling requires targets and vocab_size")
     627 +
     628 +    hidden = _encode(base_params, input_ids)
     629 +    support_indices = None
     630 +    if pinned_support:
     631 +        settled, support_indices = residual_forward(
     632 +            residual_params,
     633 +            hidden,
     634 +            num_columns=num_columns,
     635 +            top_k=top_k,
     636 +            support_router=support_router,
     637 +        )
     638 +    else:
     639 +        settled, _ = residual_forward(
     640 +            residual_params,
     641 +            hidden,
     642 +            num_columns=num_columns,
     643 +            top_k=top_k,
     644 +            support_router=support_router,
     645 +        )
     646 +
     647 +    for _ in range(1, pc_steps):
     648 +        if hep_settling_objective == "supervised_ce_gradient":
     649 +            update = _supervised_ce_hidden_update(
     650 +                base_params, settled, targets, vocab_size, max_norm=hep_update_clip_norm
     651 +            )
     652 +        elif hep_settling_objective == "prediction_entropy_gradient":
     653 +            update = _prediction_entropy_hidden_update(
     654 +                base_params, settled, max_norm=hep_update_clip_norm
     655 +            )
     656 +        elif hep_settling_objective == "temporal_consistency_gradient":
     657 +            update = _temporal_consistency_hidden_update(
     658 +                base_params, settled, max_norm=hep_update_clip_norm
     659 +            )
     660 +        else:
     661 +            proposed, _ = residual_forward(
     662 +                residual_params,
     663 +                settled,
     664 +                num_columns=num_columns,
     665 +                top_k=top_k,
     666 +                support_router=support_router,
     667 +                support_indices=support_indices,
     668 +            )
     669 +            update = _clip_update(proposed - settled, hep_update_clip_norm)
     670 +        settled = settled + jnp.float32(hep_alpha) * update
     671 +    return _decode(base_params, settled)

     674 +def _supervised_ce_hidden_update(
     675 +    base_params: dict[str, Any],
     676 +    hidden: jnp.ndarray,
     677 +    targets: jnp.ndarray,
     678 +    vocab_size: int,
     679 +    *,
     680 +    max_norm: float | None,
     681 +) -> jnp.ndarray:
     682 +    loss_fn = lambda h: _cross_entropy(_decode(base_params, h), targets, vocab_size)
     683 +    gradient = jax.grad(loss_fn)(hidden)
     684 +    return _clip_update(-gradient, max_norm)

     686 +
     687 +def _prediction_entropy_hidden_update(
     688 +    base_params: dict[str, Any],
     689 +    hidden: jnp.ndarray,
     690 +    *,
     691 +    max_norm: float | None,
     692 +) -> jnp.ndarray:
     693 +    def entropy_fn(h: jnp.ndarray) -> jnp.ndarray:
     694 +        logits = _decode(base_params, h)
     695 +        log_probs = jax.nn.log_softmax(logits, axis=-1)
     696 +        probs = jax.nn.softmax(logits, axis=-1)
     697 +        return -jnp.mean(jnp.sum(probs * log_probs, axis=-1))
     698 +
     699 +    gradient = jax.grad(entropy_fn)(hidden)
     700 +    return _clip_update(-gradient, max_norm)

     703 +def _temporal_consistency_hidden_update(
     704 +    base_params: dict[str, Any],
     705 +    hidden: jnp.ndarray,
     706 +    *,
     707 +    max_norm: float | None,
     708 +) -> jnp.ndarray:
     709 +    if hidden.shape[1] < 2:
     710 +        return jnp.zeros_like(hidden)
     711 +
     712 +    def consistency_fn(h: jnp.ndarray) -> jnp.ndarray:
     713 +        logits = _decode(base_params, h)
     714 +        teacher_probs = jax.nn.softmax(jax.lax.stop_gradient(logits[:, :-1, :]), axis=-1)
     715 +        student_log_probs = jax.nn.log_softmax(logits[:, 1:, :], axis=-1)
     716 +        return -jnp.mean(jnp.sum(teacher_probs * student_log_probs, axis=-1))
     717 +
     718 +    gradient = jax.grad(consistency_fn)(hidden)
     719 +    return _clip_update(-gradient, max_norm)
     720 +
     721 +
     722 +def _clip_update(update: jnp.ndarray, max_norm: float | None) -> jnp.ndarray:
     723 +    if max_norm is None:
     724 +        return update
     725 +    norm = jnp.linalg.norm(update, axis=-1, keepdims=True)
     726 +    scale = jnp.minimum(jnp.float32(1.0), jnp.float32(max_norm) / jnp.maximum(norm, 1e-12))
     727 +    return update * scale

     730 +def _residual_loss(
     731 +    base_params: dict[str, Any],
     732 +    residual_params: NodeParams,
     733 +    inputs: jnp.ndarray,
     734 +    targets: jnp.ndarray,
     735 +    vocab_size: int,
     736 +    *,
     737 +    objective: str,
     738 +    ce_anchor_weight: float,
     739 +    confidence_penalty_weight: float,
     740 +    margin_penalty_weight: float,
     741 +    target_logit_margin: float,
     742 +    label_smoothing_weight: float,
     743 +    focal_gamma: float,
     744 +    temporal_consistency_weight: float,
     745 +    num_columns: int,
     746 +    top_k: int,
     747 +    support_router: str,
     748 +) -> jnp.ndarray:
     749 +    logits = _forward_with_residual(
     750 +        base_params,
     751 +        residual_params,
     752 +        inputs,
     753 +        num_columns=num_columns,
     754 +        top_k=top_k,
     755 +        support_router=support_router,
     756 +    )
     757 +    prediction_logits = logits[:, :-1, :]
     758 +    prediction_targets = targets[:, :-1]
     759 +    ce_loss = _cross_entropy_from_prediction_logits(
     760 +        prediction_logits, prediction_targets, vocab_size
     761 +    )
     762 +    if objective == "supervised_ce":
     763 +        return ce_loss
     764 +    if objective == "supervised_ce_confidence_penalty":
     765 +        log_probs = jax.nn.log_softmax(prediction_logits, axis=-1)
     766 +        probs = jax.nn.softmax(prediction_logits, axis=-1)
     767 +        entropy = -jnp.mean(jnp.sum(probs * log_probs, axis=-1))
     768 +        return ce_loss - jnp.float32(confidence_penalty_weight) * entropy
     769 +    if objective == "supervised_ce_margin_penalty":
     770 +        target_logits = jnp.take_along_axis(
     771 +            prediction_logits, prediction_targets[..., None], axis=-1
     772 +        )[..., 0]
     773 +        target_mask = jax.nn.one_hot(prediction_targets, vocab_size).astype(bool)
     774 +        strongest_other = jnp.max(
     775 +            jnp.where(target_mask, jnp.asarray(-jnp.inf, dtype=prediction_logits.dtype), prediction_logits),
     776 +            axis=-1,
     777 +        )
     778 +        margin_shortfall = jax.nn.relu(
     779 +            jnp.float32(target_logit_margin) - (target_logits - strongest_other)
     780 +        )
     781 +        return ce_loss + jnp.float32(margin_penalty_weight) * jnp.mean(margin_shortfall)
     782 +    if objective == "supervised_ce_label_smoothing":
     783 +        return _label_smoothed_cross_entropy(
     784 +            prediction_logits, prediction_targets, vocab_size, label_smoothing_weight
     785 +        )
     786 +    if objective == "supervised_ce_focal":
     787 +        per_token_ce = _per_token_cross_entropy(prediction_logits, prediction_targets)
     788 +        target_probs = jnp.exp(-per_token_ce)
     789 +        return jnp.mean(jnp.power(1.0 - target_probs, focal_gamma) * per_token_ce)
     790 +    if objective == "supervised_ce_temporal_consistency":
     791 +        if prediction_logits.shape[1] < 2:
     792 +            return ce_loss
     793 +        teacher_probs = jax.nn.softmax(
     794 +            jax.lax.stop_gradient(prediction_logits[:, :-1, :]), axis=-1
     795 +        )
     796 +        student_log_probs = jax.nn.log_softmax(prediction_logits[:, 1:, :], axis=-1)
     797 +        temporal_loss = -jnp.mean(jnp.sum(teacher_probs * student_log_probs, axis=-1))
     798 +        return ce_loss + jnp.float32(temporal_consistency_weight) * temporal_loss
     799 +    if objective in {"pc_logit_mse", "pc_logit_mse_ce_anchor"}:
     800 +        probs = jax.nn.softmax(prediction_logits, axis=-1)
     801 +        target_probs = jax.nn.one_hot(prediction_targets, vocab_size, dtype=probs.dtype)
     802 +        pc_loss = jnp.mean(jnp.square(probs - target_probs))
     803 +        if objective == "pc_logit_mse":
     804 +            return pc_loss
     805 +        return pc_loss + jnp.float32(ce_anchor_weight) * ce_loss
     806 +    raise ValueError(f"Unsupported residual objective: {objective}")

     809 +def _cross_entropy(logits: jnp.ndarray, targets: jnp.ndarray, vocab_size: int) -> jnp.ndarray:
     810 +    return _cross_entropy_from_prediction_logits(logits[:, :-1, :], targets[:, :-1], vocab_size)

     813 +def _cross_entropy_from_prediction_logits(
     814 +    prediction_logits: jnp.ndarray,
     815 +    prediction_targets: jnp.ndarray,
     816 +    vocab_size: int,
     817 +) -> jnp.ndarray:
     818 +    _ = vocab_size
     819 +    return jnp.mean(_per_token_cross_entropy(prediction_logits, prediction_targets))

     821 +
     822 +def _per_token_cross_entropy(
     823 +    prediction_logits: jnp.ndarray,
     824 +    prediction_targets: jnp.ndarray,
     825 +) -> jnp.ndarray:
     826 +    log_probs = jax.nn.log_softmax(prediction_logits, axis=-1)
     827 +    return -jnp.take_along_axis(log_probs, prediction_targets[..., None], axis=-1)[..., 0]

     830 +def _label_smoothed_cross_entropy(
     831 +    prediction_logits: jnp.ndarray,
     832 +    prediction_targets: jnp.ndarray,
     833 +    vocab_size: int,
     834 +    smoothing: float,
     835 +) -> jnp.ndarray:
     836 +    log_probs = jax.nn.log_softmax(prediction_logits, axis=-1)
     837 +    targets = jax.nn.one_hot(prediction_targets, vocab_size, dtype=log_probs.dtype)
     838 +    smooth = jnp.float32(smoothing)
     839 +    targets = targets * (1.0 - smooth) + smooth / vocab_size
     840 +    return -jnp.mean(jnp.sum(targets * log_probs, axis=-1))

     937 +def _hep_support_instability(
     938 +    base_params: dict[str, Any],
     939 +    residual_params: NodeParams,
     940 +    inputs: jnp.ndarray,
     941 +    *,
     942 +    pc_steps: int,
     943 +    hep_alpha: float,
     944 +    num_columns: int,
     945 +    top_k: int,
     946 +    support_router: str,
     947 +    hep_update_clip_norm: float | None = None,
     948 +) -> dict[str, float | int | bool | None]:
     949 +    if pc_steps < 1:
     950 +        raise ValueError("pc_steps must be at least 1")
     951 +    if hep_alpha < 0.0 or hep_alpha > 1.0:
     952 +        raise ValueError("hep_alpha must be between 0.0 and 1.0")
     953 +    hidden = _encode(base_params, inputs)
     954 +    repicked_settled, initial_support = residual_forward(
     955 +        residual_params,
     956 +        hidden,
     957 +        num_columns=num_columns,
     958 +        top_k=top_k,
     959 +        support_router=support_router,
     960 +    )
     961 +    pinned_settled = repicked_settled
     962 +    changed_positions = jnp.zeros(initial_support.shape[:-1], dtype=bool)
     963 +    support_transition_count = 0
     964 +    for _ in range(1, pc_steps):
     965 +        repicked_proposed, repicked_support = residual_forward(
     966 +            residual_params,
     967 +            repicked_settled,
     968 +            num_columns=num_columns,
     969 +            top_k=top_k,
     970 +            support_router=support_router,
     971 +        )
     972 +        pinned_proposed, _ = residual_forward(
     973 +            residual_params,
     974 +            pinned_settled,
     975 +            num_columns=num_columns,
     976 +            top_k=top_k,
     977 +            support_router=support_router,
     978 +            support_indices=initial_support,
     979 +        )
     980 +        repicked_update = _clip_update(
     981 +            repicked_proposed - repicked_settled, hep_update_clip_norm
     982 +        )
     983 +        pinned_update = _clip_update(
     984 +            pinned_proposed - pinned_settled, hep_update_clip_norm
     985 +        )
     986 +        repicked_settled = repicked_settled + jnp.float32(hep_alpha) * repicked_update
     987 +        pinned_settled = pinned_settled + jnp.float32(hep_alpha) * pinned_update
     988 +        changed = jnp.any(repicked_support != initial_support, axis=-1)
     989 +        support_transition_count += int(jnp.sum(changed))
     990 +        changed_positions = jnp.logical_or(changed_positions, changed)
     991 +
     992 +    total_positions = int(changed_positions.size)
     993 +    changed_position_count = int(jnp.sum(changed_positions))
     994 +    repicked_logits = _decode(base_params, repicked_settled)
     995 +    pinned_logits = _decode(base_params, pinned_settled)
     996 +    return {
     997 +        "pc_steps": pc_steps,
     998 +        "hep_alpha": float(hep_alpha),
     999 +        "hep_update_clip_norm": hep_update_clip_norm,
    1000 +        "support_positions": total_positions,
    1001 +        "support_changed_positions": changed_position_count,
    1002 +        "support_change_fraction": (
    1003 +            changed_position_count / total_positions if total_positions else 0.0
    1004 +        ),
    1005 +        "support_transition_count": support_transition_count,
    1006 +        "pinned_vs_repicked_logit_delta": _max_abs(pinned_logits - repicked_logits),
    1007 +        "support_changed": changed_position_count > 0,
    1008 +    }

    1097 +def _apply_hep_support_stress(
    1098 +    base_params: dict[str, Any],
    1099 +    residual_params: NodeParams,
    1100 +    inputs: jnp.ndarray,
    1101 +    *,
    1102 +    num_columns: int,

    1109 +    hidden = _encode(base_params, inputs)

    1117 +    atom_values = atom_values.at[1, 0, :].set(-0.01 * direction)
    1118 +    weights = dict(residual_params.weights)
    1119 +    weights["column_score_w"] = w
    1120 +    weights["atom_values"] = atom_values
